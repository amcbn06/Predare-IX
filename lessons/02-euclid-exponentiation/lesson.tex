\section{Algoritmul lui Euclid}

\begin{frame}{Cel mai mare divizor comun}
  \begin{block}{Cel mai mare divizor comun}
    $\text{cmmdc}(a, b)$ = cel mai mare număr natural $d$ astfel încât $d \mid a$ și $d \mid b$. Se mai notează și $\gcd(a, b)$ (\textit{greatest common divisor}) sau doar $(a, b)$.
  \end{block}
  \bigskip
  \begin{block}{Cel mai mic multiplu comun}
    $\text{cmmmc}(a, b)$ = cel mai mic număr natural $m$ astfel încât $a \mid m$ și $b \mid m$. Se mai notează și $\text{lcm}(a, b)$ (\textit{lowest common multiple}) sau doar $[a, b]$.
  \end{block}
  \bigskip
  Exemple:
  \begin{itemize}
    \item $\gcd(12, 8) = 4$ \quad $\gcd(17, 5) = 1$ \quad $\gcd(100, 75) = 25$
    \item $\text{lcm}(4, 6) = 12$ \quad $\text{lcm}(3, 5) = 15$
  \end{itemize}
\end{frame}

\begin{frame}{Proprietăți (I)}
  \begin{itemize}
    \item $\gcd(a, 0) = \gcd(a, a) = |a|$ \quad \quad (\textit{cazuri de bază})
    \item $\gcd(a, b) = \gcd(b, a)$ \quad (\textit{comutativitate})
    \item $\gcd(a, \gcd(b, c)) = \gcd(\gcd(a, b), c)$ \quad (\textit{asociativitate})
    \item $\gcd(k \cdot a, k \cdot b) = k \cdot \gcd(a, b)$, pentru $k > 0$
    \item dacă $\gcd(a, b) = d$, atunci $\gcd\!\left(\dfrac{a}{d},\ \dfrac{b}{d}\right) = 1$
    \item $a \cdot b = \gcd(a, b) \cdot \text{lcm}(a, b)$
    \item dacă $\gcd(a, b) = 1$, spunem că $a$ și $b$ sunt \textit{prime între ele} (\textit{coprime})
  \end{itemize}
\end{frame}

\begin{frame}{Proprietăți (II)}
  \begin{itemize}
    \item $\gcd(a, b) = \gcd(a \pm k \cdot b,\ b)$, $\forall\, k \in \mathbb{N}$
    \item \textbf{Consecință directă:} $\gcd(a, b) = \gcd(b,\ a \bmod b)$ \\
      {\footnotesize\textcolor{gray}{$a \bmod b$ = restul împărțirii lui $a$ la $b$}}
    \item \textbf{Prin descompunere în factori primi:} dacă
      $n = \prod p_i^{a_i}$ și $m = \prod p_i^{b_i}$, atunci:
      \[
        \gcd(n, m) = \prod p_i^{\min(a_i,\, b_i)} \qquad
        \text{lcm}(n, m) = \prod p_i^{\max(a_i,\, b_i)}
      \]
      Exemplu: $n = 2^3 \cdot 3^2,\ m = 2^1 \cdot 3^4 \;\Rightarrow\;
        \gcd = 2^1 \cdot 3^2 = 18,\;
        \text{lcm} = 2^3 \cdot 3^4 = 648$
  \end{itemize}
\end{frame}

\begin{frame}{Algoritmul lui Euclid \textemdash{} Idee}
  Știm că $\gcd(a, b) = \gcd(a - b, b)$. Putem aplica această regulă repetat până ajungem la cazul de bază $\gcd(a, 0) = a$.
  \bigskip

  Exemplu: $\gcd(12, 8)$
  \begin{itemize}
    \item $\gcd(12, 8) = \gcd(4, 8) = \gcd(4, 4) = \gcd(0, 4) = 4$ \checkmark
  \end{itemize}
  \bigskip

  \begin{block}{\textbf{\textcolor{red}{Contraexemplu:}} $a = 1000, b = 1$}
    $\gcd(1000, 1) \rightarrow \gcd(999, 1) \rightarrow \gcd(998, 1) \rightarrow \cdots \rightarrow \gcd(0, 1) = 1$\\
    \medskip
    1000 de pași pentru un rezultat evident. Putem face mai bine?
  \end{block}
\end{frame}

\begin{frame}{Algoritmul lui Euclid \textemdash{} Idee}
  Observăm că scăderile repetate cu același $b$ sunt echivalente cu restul împărțirii:
  \[
    \gcd(a, b) = \gcd(a - b, b) = \gcd(a - 2b, b) = \cdots = \gcd(a \bmod b,\ b)
  \]
  \begin{block}{Observație cheie}
    $\gcd(a, b) = \gcd(b,\ a \bmod b)$, \quad cu cazul de bază $\gcd(a, 0) = a$.
  \end{block}
  \bigskip
  Exemplu: $\gcd(48, 18)$
  \begin{itemize}
    \item $\gcd(48, 18) = \gcd(18, 12)$ \quad ($48 \bmod 18 = 12$)
    \item $\gcd(18, 12) = \gcd(12, 6)$  \quad ($18 \bmod 12 = 6$)
    \item $\gcd(12,\ 6) = \gcd(6,\ 0)$  \quad\; ($12 \bmod 6 = 0$)
    \item $\gcd(6, 0) = 6$ \checkmark
  \end{itemize}
\end{frame}

\begin{frame}[fragile, shrink]{Algoritmul lui Euclid \textemdash{} Implementare iterativă}
  \begin{center}
  \begin{lstlisting}
int gcd(int a, int b) {
    while (b != 0) {
        int r = a % b;
        a = b;
        b = r;
    }
    return a;
}
  \end{lstlisting}
  \end{center}
  \begin{itemize}
    \item Timp: $O(\log \min(a, b))$ \quad Spațiu: $O(1)$
    \item Exemplu: \lstinline[keepspaces=true]|gcd(48, 18) = 6|
    \item Cel mai mic multiplu comun: \lstinline[keepspaces=true]|lcm(a, b) = a / gcd(a, b) * b| \\
      {\footnotesize\textcolor{gray}{Tip: împarte înainte de înmulțire pentru a evita overflow}}
  \end{itemize}
\end{frame}

\section{Algoritmul lui Euclid extins}

\begin{frame}{Algoritmul lui Euclid extins \textemdash{} Idee}
  \begin{block}{Identitatea lui Bézout}
    Pentru orice $a, b \in \mathbb{Z}$, există $x, y \in \mathbb{Z}$ astfel încât:
    \[ a \cdot x + b \cdot y = \gcd(a, b) \]
  \end{block}
  \medskip
  Vrem să rezolvăm $a \cdot x + b \cdot y = g$. Scriem $a = q \cdot b + r$, unde $q = \lfloor a / b \rfloor$, $r = a \bmod b$:
  \[
    (\underbrace{q \cdot b + r}_{a}) \cdot x + b \cdot y = g
    \;\;\Rightarrow\;\;
    b \cdot (q \cdot x + y) + r \cdot x = g
  \]
  \vfill
  {\footnotesize\textcolor{gray}{\textit{$\triangleright$ Această secțiune este un concept ușor mai dificil \textemdash{} poți reveni la ea după ce ești confortabil cu restul lecției.}}}
\end{frame}

\begin{frame}{Algoritmul lui Euclid extins \textemdash{} Idee}
  Notăm $x' = q \cdot x + y$ și $y' = x$. Obținem:
  \[
    b \cdot x' + r \cdot y' = g
  \]
  \begin{block}{Observație cheie}
    Aceeași formă ca înainte, cu $(b,\ r)$ în loc de $(a,\ b)$ \textemdash{} exact un pas din algoritmul lui Euclid!
  \end{block}
  \medskip
  Putem folosi metoda mersului invers pentru o soluție iterativă sau putem aplica recursiv până la cazul de bază $b = 0$:
  \[
    g \cdot x + 0 \cdot y = g \;\Rightarrow\; x = 1,\ y = 0
  \]
  Reconstruim soluția înapoi la fiecare pas: $x = y'$,\quad $y = x' - q \cdot y'$.
\end{frame}

\begin{frame}[shrink]{Exemplu pe hârtie: $22x + 13y = \gcd(22, 13)$}
  \textbf{Pasul 1: Aplicăm algoritmul lui Euclid}
  \begin{itemize}
    \item $22 = 1 \cdot 13 + 9$ \hfill {\footnotesize\textcolor{gray}{$\Rightarrow 9 = 22 - 1 \cdot 13$}}
    \item $13 = 1 \cdot 9 + 4$ \hfill {\footnotesize\textcolor{gray}{$\Rightarrow 4 = 13 - 1 \cdot 9$}}
    \item $9 = 2 \cdot 4 + 1$ \hfill {\footnotesize\textcolor{gray}{$\Rightarrow 1 = 9 - 2 \cdot 4$}}
    \item $4 = 4 \cdot 1 + 0 \;\Rightarrow\; \gcd(22, 13) = 1$
  \end{itemize}

  \bigskip
  \textbf{Pasul 2: Înlocuim înapoi}
  \begin{itemize}
    \item $1 = 1 \cdot \bred{9} - 2 \cdot \bblu{4}$
    \item $1 = 1 \cdot \bred{9} - 2 \cdot (\bblu{13} - 1 \cdot \bred{9}) = 3 \cdot \bred{9} - 2 \cdot \bblu{13}$
    \item $1 = 3 \cdot (1 \cdot \bred{22} - 1 \cdot \bblu{13}) - 2 \cdot \bblu{13} = 3 \cdot \bred{22} - 5 \cdot \bblu{13}$
  \end{itemize}
  \bigskip
  \begin{block}{Soluție}
    $x = 3,\ y = -5$ \quad \checkmark \quad ($22 \cdot 3 + 13 \cdot (-5) = 66 - 65 = 1$)
  \end{block}
\end{frame}

\begin{frame}[fragile, shrink]{Algoritmul lui Euclid extins \textemdash{} Implementare iterativă}
  \begin{center}
  \begin{lstlisting}
// Gaseste x, y cu: a*x + b*y = gcd(a, b)
int gcd_ext(int a, int b, int &x, int &y) {
    int x0 = 1, y0 = 0;
    int x1 = 0, y1 = 1;
    while (b != 0) {
        int q  = a / b;
        int xn = x0 - q * x1;
        int yn = y0 - q * y1;
        int r  = a % b;
        a = b;  b = r;
        x0 = x1; y0 = y1;
        x1 = xn; y1 = yn;
    }
    x = x0; y = y0;
    return a;
}
  \end{lstlisting}
  \end{center}
  \begin{itemize}
    \item \lstinline|int &x| = parametru prin referință: modificările din funcție se văd la apelant
    \item Timp: $O(\log \min(a, b))$ \quad Spațiu: $O(1)$
  \end{itemize}
\end{frame}

\section{Exponențiere rapidă}

\begin{frame}{Exponențiere rapidă \textemdash{} Idee}
  \begin{block}{Problema}
    Calculează $a^n$ eficient. Varianta naivă ($a \cdot a \cdots a$, de $n$ ori) are complexitate $O(n)$.
  \end{block}
  \bigskip
  \begin{block}{Observație cheie}
    \[
      a^n = \begin{cases}
        1 & n = 0 \\
        \left(a^{n/2}\right)^2 & n \text{ par} \\
        a \cdot a^{n-1} & n \text{ impar}
      \end{cases}
    \]
  \end{block}
  \bigskip
  Exemplu: $2^{10} \rightarrow 2^5 \rightarrow 2^4 \rightarrow 2^2 \rightarrow 2^1 \rightarrow 2^0$ \quad (5 pași în loc de 10)
\end{frame}

\begin{frame}[fragile, shrink]{Exponențiere rapidă \textemdash{} Implementare iterativă}
  \begin{center}
  \begin{lstlisting}
long long power(long long base, int exp) {
    long long result = 1;
    while (exp > 0) {
        if (exp % 2 == 1)
            result *= base;
        base *= base;
        exp /= 2;
    }
    return result;
}
  \end{lstlisting}
  \end{center}
  \begin{itemize}
    \item Timp: $O(\log n)$ \quad Spațiu: $O(1)$
    \item \textbf{Atenție:} \lstinline|base * base| poate da overflow pentru baze mari \textemdash{} în concursuri se lucrează de obicei modulo un număr prim
  \end{itemize}
\end{frame}

\section{Bonus \textemdash{} Implementări recursive}

\begin{frame}{Recursivitate \textemdash{} Ce este?}
  \begin{block}{Definiție}
    O funcție este \textbf{recursivă} dacă se apelează pe ea însăși.
  \end{block}
  \bigskip
  Orice funcție recursivă are nevoie de:
  \begin{itemize}
    \item Un \textbf{caz de bază} \textemdash{} condiția de oprire (altfel rulează la infinit)
    \item Un \textbf{apel recursiv} pe o subproblemă mai mică
  \end{itemize}
  \bigskip
  Exemplu: $n! = n \cdot (n-1)!$, cu cazul de bază $0! = 1$
  \beginnote{Recursivitate \textemdash{} concept nou}
\end{frame}

\begin{frame}{De ce implementări recursive?}
  \begin{itemize}
    \item Codul reflectă direct definiția matematică \textemdash{} mai ușor de înțeles și de scris
    \item De obicei mai scurt decât varianta iterativă
  \end{itemize}
  \medskip
  \begin{block}{\textbf{\textcolor{red}{Atenție}} \textemdash{} dezavantaje}
    \begin{itemize}
      \item Overhead de apeluri de funcție \textemdash{} ușor mai lent
      \item Pentru intrări foarte mari poate da \textbf{stack overflow}
    \end{itemize}
  \end{block}
  \medskip
  \begin{block}{Concluzie}
    Varianta recursivă e utilă pentru a \textbf{înțelege} algoritmul; \\ varianta iterativă e de preferat în concursuri.
  \end{block}
\end{frame}

\begin{frame}[fragile, shrink]{Algoritmul lui Euclid \textemdash{} Implementare recursivă}
  \begin{center}
  \begin{lstlisting}
int gcd(int a, int b) {
    if (b == 0)
        return a;
    return gcd(b, a % b);
}
  \end{lstlisting}
  \end{center}
  \bigskip
  \begin{itemize}
    \item Codul reflectă direct proprietatea $\gcd(a, b) = \gcd(b,\ a \bmod b)$
    \item Aceeași complexitate ca varianta iterativă: $O(\log \min(a, b))$
  \end{itemize}
\end{frame}

\begin{frame}[fragile, shrink]{Algoritmul lui Euclid extins \textemdash{} Implementare recursivă}
  \begin{center}
  \begin{lstlisting}
int gcd_ext(int a, int b, int &x, int &y) {
    if (b == 0) {
        x = 1; y = 0;
        return a;
    }
    int x1, y1;
    int d = gcd_ext(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return d;
}
  \end{lstlisting}
  \end{center}
  \begin{itemize}
    \item Timp: $O(\log \min(a, b))$ \quad Spațiu: $O(\log \min(a, b))$ \textemdash{} stiva de apeluri
  \end{itemize}
\end{frame}

\begin{frame}[fragile, shrink]{Exponențiere rapidă \textemdash{} Implementare recursivă}
  \begin{center}
  \begin{lstlisting}
long long power(long long base, int exp) {
    if (exp == 0)
        return 1;
    if (exp % 2 == 0) {
        long long half = power(base, exp / 2);
        return half * half;
    }
    return base * power(base, exp - 1);
}
  \end{lstlisting}
  \end{center}
  \begin{itemize}
    \item \lstinline|half * half| evită calculul dublu \lstinline[keepspaces=true]|power(base, exp / 2) * power(base, exp / 2)|
    \item Timp: $O(\log n)$ \quad Spațiu: $O(\log n)$ \textemdash{} stiva de apeluri
  \end{itemize}
\end{frame}
